What is Currying?

Currying is a functional programming technique...

class Student{
  constructor(name,age,marks){
    this.name=name;
    this.age=age;
    this.marks=marks;   
  }

  setPlacementAge(minAge) {
    return (minMarks) => this.marks >= minMarks && this.age >= minAge;
  }
}

function createNewStudents(name, age, marks){
  const Riya = new Student(name, age, marks);
  console.log(Riya.setPlacementAge(18)(40))
}
createNewStudents('Riya', 18, 41); // output: true;
